### 1. Master Startup Script (`start-aims.sh`)

**Location**: Project root directory
**Purpose**: Orchestrates both backend and frontend startup

```bash
#!/bin/bash

# AIMS Project Master Startup Script
# =================================

set -e  # Exit on any error

PROJECT_ROOT="/Users/namu10x/workspace/hust/javafxml-AIMS"
LOG_DIR="$PROJECT_ROOT/logs"
BACKEND_LOG="$LOG_DIR/backend.log"
FRONTEND_LOG="$LOG_DIR/frontend.log"

# Colors for output
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m' # No Color

# Create logs directory
mkdir -p "$LOG_DIR"

echo -e "${BLUE}üöÄ AIMS Project Startup Automation${NC}"
echo -e "${BLUE}===================================${NC}"
echo ""

# Function to check if port is available
check_port() {
    local port=$1
    if lsof -Pi :$port -sTCP:LISTEN -t >/dev/null; then
        echo -e "${YELLOW}‚ö†Ô∏è  Port $port is already in use${NC}"
        return 1
    fi
    return 0
}

# Function to wait for service to be ready
wait_for_service() {
    local url=$1
    local service_name=$2
    local max_attempts=30
    local attempt=1
    
    echo -e "${YELLOW}üîç Waiting for $service_name to be ready...${NC}"
    
    while [ $attempt -le $max_attempts ]; do
        if curl -s "$url" > /dev/null 2>&1; then
            echo -e "${GREEN}‚úÖ $service_name is ready!${NC}"
            return 0
        fi
        echo -n "."
        sleep 2
        ((attempt++))
    done
    
    echo -e "${RED}‚ùå $service_name failed to start within timeout${NC}"
    return 1
}

# Pre-flight checks
echo -e "${BLUE}üîç Pre-flight System Checks${NC}"
echo "================================"

# Check Java
if command -v java &> /dev/null; then
    JAVA_VERSION=$(java -version 2>&1 | head -n 1 | cut -d'"' -f2)
    echo -e "${GREEN}‚úÖ Java: $JAVA_VERSION${NC}"
else
    echo -e "${RED}‚ùå Java not found${NC}"
    exit 1
fi

# Check Maven
if command -v mvn &> /dev/null; then
    MVN_VERSION=$(mvn -version | head -n 1 | cut -d' ' -f3)
    echo -e "${GREEN}‚úÖ Maven: $MVN_VERSION${NC}"
else
    echo -e "${RED}‚ùå Maven not found${NC}"
    exit 1
fi

# Check Node.js
if command -v node &> /dev/null; then
    NODE_VERSION=$(node --version)
    echo -e "${GREEN}‚úÖ Node.js: $NODE_VERSION${NC}"
else
    echo -e "${RED}‚ùå Node.js not found${NC}"
    exit 1
fi

# Check npm
if command -v npm &> /dev/null; then
    NPM_VERSION=$(npm --version)
    echo -e "${GREEN}‚úÖ npm: $NPM_VERSION${NC}"
else
    echo -e "${RED}‚ùå npm not found${NC}"
    exit 1
fi

# Check ports
echo ""
echo -e "${BLUE}üîå Port Availability Check${NC}"
echo "=========================="

if check_port 8080; then
    echo -e "${GREEN}‚úÖ Port 8080 (Backend) is available${NC}"
else
    echo -e "${RED}‚ùå Port 8080 is occupied. Please free it before starting.${NC}"
    exit 1
fi

if check_port 3000; then
    echo -e "${GREEN}‚úÖ Port 3000 (Frontend) is available${NC}"
else
    echo -e "${YELLOW}‚ö†Ô∏è  Port 3000 is occupied. Vite will auto-increment.${NC}"
fi

echo ""
echo -e "${BLUE}üé¨ Starting Services${NC}"
echo "==================="

# Start Backend in new terminal
echo -e "${YELLOW}üîß Starting Backend (Spring Boot)...${NC}"
osascript -e "
tell application \"Terminal\"
    do script \"cd '$PROJECT_ROOT' && ./scripts/start-backend.sh\"
end tell
"

# Wait a moment for backend to initialize
sleep 5

# Start Frontend in new terminal
echo -e "${YELLOW}üé® Starting Frontend (React + Vite)...${NC}"
osascript -e "
tell application \"Terminal\"
    do script \"cd '$PROJECT_ROOT' && ./scripts/start-frontend.sh\"
end tell
"

# Wait for services to be ready
echo ""
echo -e "${BLUE}‚è≥ Service Health Checks${NC}"
echo "======================="

# Wait for backend
if wait_for_service "http://localhost:8080/actuator/health" "Backend API"; then
    echo -e "${GREEN}üéâ Backend is operational!${NC}"
else
    echo -e "${RED}‚ùå Backend startup failed${NC}"
    exit 1
fi

# Wait for frontend
if wait_for_service "http://localhost:3000" "Frontend App"; then
    echo -e "${GREEN}üéâ Frontend is operational!${NC}"
else
    echo -e "${RED}‚ùå Frontend startup failed${NC}"
    exit 1
fi

echo ""
echo -e "${GREEN}üéâ AIMS Project Successfully Started!${NC}"
echo -e "${GREEN}====================================${NC}"
echo ""
echo -e "${BLUE}üì± Access URLs:${NC}"
echo -e "   üåê Frontend:     ${GREEN}http://localhost:3000${NC}"
echo -e "   üîß Backend API:  ${GREEN}http://localhost:8080${NC}"
echo -e "   üìö API Docs:     ${GREEN}http://localhost:8080/swagger-ui.html${NC}"
echo -e "   ‚ù§Ô∏è  Health:      ${GREEN}http://localhost:8080/actuator/health${NC}"
echo ""
echo -e "${BLUE}üìù Logs:${NC}"
echo -e "   üìÑ Backend:      ${YELLOW}tail -f $BACKEND_LOG${NC}"
echo -e "   üìÑ Frontend:     ${YELLOW}tail -f $FRONTEND_LOG${NC}"
echo ""
echo -e "${YELLOW}üí° To stop services, close the respective terminal windows${NC}"
```

### 4. Stop Services Script (`stop-aims.sh`)

```bash
#!/bin/bash

# AIMS Project Stop Script
# =======================

# Colors
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m'

echo -e "${BLUE}üõë Stopping AIMS Project Services${NC}"
echo -e "${BLUE}=================================${NC}"
echo ""

# Function to kill process on port
kill_port() {
    local port=$1
    local service_name=$2
    
    local pid=$(lsof -ti:$port)
    if [ ! -z "$pid" ]; then
        echo -e "${YELLOW}üîç Found $service_name running on port $port (PID: $pid)${NC}"
        kill -TERM $pid
        sleep 2
        
        # Force kill if still running
        if kill -0 $pid 2>/dev/null; then
            echo -e "${YELLOW}üí™ Force killing $service_name...${NC}"
            kill -KILL $pid
        fi
        
        echo -e "${GREEN}‚úÖ $service_name stopped${NC}"
    else
        echo -e "${YELLOW}‚ÑπÔ∏è  $service_name not running on port $port${NC}"
    fi
}

# Stop backend (port 8080)
kill_port 8080 "Backend API"

# Stop frontend (port 3000)
kill_port 3000 "Frontend Dev Server"

# Also check common alternative ports
kill_port 3001 "Frontend Dev Server (Alt)"

echo ""
echo -e "${GREEN}üéâ All AIMS services stopped successfully!${NC}"
```
